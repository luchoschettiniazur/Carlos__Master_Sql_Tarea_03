--Crear procedure crear en bbdd
--**************************************************************************
CREATE PROCEDURE spDirUsuariosPorAnioCreacion
	@anioInicio int,
	@anioFin int
AS
select u.FechaCreacion, u.UsuarioId, u.Nombre, u.Apellido, ux.Calle, ux.CodigoPostal, c.Nombre as Ciudad, p.Nombre as Pais
from Usuario as u
join UsuarioExtra as ux on u.UsuarioId = ux.UsuarioId
join Ciudad as c on  ux.CiudadId = c.CiudadId 
join Pais as p on c.PaisId = p.PaisId
where year(u.FechaCreacion) >= @anioInicio and year(u.FechaCreacion) <= @aniofin
order by u.FechaCreacion
go

--consultas de acumulados:  para tarea 03
--**************************************************************************
--Count
select p.Nombre as Pais, Count(*) as TotalUsuariosPorPais
from Usuario as u
join UsuarioExtra as ux on u.UsuarioId = ux.UsuarioId
join Ciudad as c on  ux.CiudadId = c.CiudadId 
join Pais as p on c.PaisId = p.PaisId
where year(u.FechaCreacion) >= 2021 and year(u.FechaCreacion) <= 2023
group by p.Nombre

--Min
select u.FechaCreacion as FechaCreacionUsuario, Min(ux.PuntosAcumulados) as PuntosMinimosPorFechaCreacion
from Usuario as u
join UsuarioExtra as ux on u.UsuarioId = ux.UsuarioId
where year(u.FechaCreacion) >= 2022 and year(u.FechaCreacion) <= 2023
group by u.FechaCreacion

--Max
select u.FechaCreacion as FechaCreacionUsuario, Max(ux.PuntosAcumulados) as PuntosMaximoPorFechaCreacion
from Usuario as u
join UsuarioExtra as ux on u.UsuarioId = ux.UsuarioId
where year(u.FechaCreacion) >= 2022 and year(u.FechaCreacion) <= 2023
group by u.FechaCreacion
--Avg
select u.FechaCreacion as FechaCreacionUsuario, AVG(ux.PuntosAcumulados) as PuntosMediosPorFechaCreacion
from Usuario as u
join UsuarioExtra as ux on u.UsuarioId = ux.UsuarioId
where year(u.FechaCreacion) >= 2022 and year(u.FechaCreacion) <= 2023
group by u.FechaCreacion
